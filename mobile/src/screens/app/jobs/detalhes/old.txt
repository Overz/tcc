  const updateConditions = () => {
    const {
      route: { params },
    } = (props as unknown) as {
      route: {
        params: {
          isContratado: boolean;
          isContratante: boolean;
          showRenewDate: boolean;
          showCandidates: boolean;
        };
      };
    };

    if (job && params) {
      const publicacao = new Date(job?.dtvalidade);
      const vencimento = new Date(publicacao.getTime() + ms('7d'));
      const isValid = vencimento > new Date();
      const temContratado =
        job.contratado !== null && job.contratado !== undefined;
      let canFinish = false;
      let canRenewDate = false;
      let showCandidates = false; // Candidatos
      let renewDate = false; // Pode renovar Data?
      let btnName = 'Candidatar-se'; // Mensagem padrão
      let color = colors.blueSky;
      let icon = () => <AntDesign name="solution1" color="white" size={24} />;

      setConditions((c) => ({
        ...c,
        isContratado: params.isContratado,
        isContratante: params.isContratante,
      }));

      if (params.isContratante && temContratado && isValid) {
        color = colors.greyLight;
        btnName = 'Cancelar';
        showCandidates = true;
        canRenewDate = true;
        icon = () => <MaterialIcons name="cancel" size={24} color="white" />;
      }

      if (params.isContratante && temContratado && isValid) {
        // Data válida para cancelar ?
        btnName = 'Cancelar';
        color = colors.black;
        icon = () => <MaterialIcons name="cancel" size={24} color="white" />;
        showCandidates = true;
        canRenewDate = true;
      }

      if (params.isContratante && temContratado && !isValid) {
        canFinish = true;
        btnName = 'Finalizar';
        renewDate = true;
        showCandidates = false;
        canRenewDate = true;
        icon = () => <AntDesign name="checkcircleo" size={24} color="white" />;
      }

      console.log({
        showCandidates,
        canFinish,
        renewDate,
        canRenewDate,
        isContratante: params.isContratante,
        isContratado: params.isContratado,
        temContratado: job.contratado !== null && job.contratado !== undefined,
        isValid,
      });

      if (params.isContratado) {
        canFinish = true;
        btnName = 'Finalizar';
        icon = () => <AntDesign name="checkcircleo" size={24} color="white" />;
      }

      setConditions((c) => ({
        ...c,
        isValid,
        showCandidates,
        renewDate,
        btnName,
        color,
        icon,
        canFinish,
        canRenewDate,
        isFinished: job.status === 'F' || job.status === 'C',
      }));
    }
    setLoadScreen(false);
  };
